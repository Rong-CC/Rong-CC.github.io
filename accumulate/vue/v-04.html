    # vueCli3.0+TypeScript的项目搭建

    1.安装typescript 和 vue-cli3.0 的环境

    npm i -g @vue/cli typescript
如果之前安装了vue 2.96 请卸载之前的环境
npm uninstall vue-cli -g

    2.创建项目

    vue create vue-ts   ## webpack
npm init @vitejs/app vue-app --template vue-ts  ## vite

    3、新增route

    npm install vue-router@4 --save

    创建一个router目录 创建index.ts 配置路由

    import { createRouter, createWebHashHistory, RouteRecordRaw } from 'vue-router'

const routes: Array<RouteRecordRaw> = [
  {
    path: '/',
    name: 'Home',
    meta: {
      title: '首页',
      keepAlive: true,
    },
    component: () => import('../view/index.vue'),
  },
]
const router = createRouter({
  history: createWebHashHistory(),
  routes,
})

export default router

    在main.ts挂载路由

    import { createApp } from 'vue'
import App from './App.vue'
import router from "./router";
createApp(App)
.use(router)
.mount('#app')

    4、创建vuex

    npm install vuex@next --save

    创建store 目录 index.ts文件配置

    import { createStore } from 'vuex'
const store = createStore({
  state: {
    list: [],
  },
  getters: {},
  mutations: {},
  actions: {},
  modules:{}
})
export default store

    main.ts 挂载vuex

    import { createApp } from 'vue'
import ElementPlus from 'element-plus';
import router from './router'
import store from './store'
import 'element-plus/lib/theme-chalk/index.css';
import App from './App.vue'

createApp(App)
.use(router)
.use(store)
.use(ElementPlus)
.mount('#app')

    5、代码环境变量配置

    src目录中创建.env.development 开发环境配置

    VUE_APP_BASE_ENV = 'dev'

    src 目录中创建 .env.test 测试环境配置

    VUE_APP_BASE_ENV = 'test'

    src目录中创建 .env.production 生产环境配置

    VUE_APP_BASE_ENV = 'prod'

    6、请求拦截

    npm install axios --save

    src目录创建utils目录

    utils目录创建request.ts

    import axios, { AxiosRequestConfig, AxiosResponse } from 'axios'
import { ElMessage } from 'element-plus'
const service = axios.create({
  baseURL: process.env.VUE_APP_BASE_ENV,
  timeout: 5000,
})
// 请求拦截
service.interceptors.request.use(
  (config: AxiosRequestConfig) => {
    config.headers.token = ''
    return config
  },
  (error: any) => {
    // 错误处理
    return Promise.reject(error)
  }
)
service.interceptors.response.use(
  (response: AxiosResponse) => {
    const res = response.data
    if (res.code !== 200 || !res.success) {
      //错误code 统一拦截
      ElMessage.error(response.data.msg)
      return Promise.reject(new Error(response.data.msg))
    } else {
      return res
    }
  },
  (error) => {
    ElMessage({
      message: error.msg,
      type: 'error',
      duration: 5 * 1000
   })
    return Promise.reject(error)
  }
)
export default service
