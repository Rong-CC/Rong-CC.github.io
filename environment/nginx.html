    # Linux中安装nginx

    nginx 常用的工具

    yum -y install gcc gcc-c++ autoconf pcre-devel make automake
yum -y install wget httpd-tools vim

    通过yum 源来命令下载 yum install nginx

    nginx -v 查看 nginx 是否存在

    nginx -t 检查nginx 的语法是否存在错误

    ngnix -s reload 重新载入配置文件

    systemctl restart nginx.service 重启nginx

    nginx -s stop 停止服务

    nginx -s quit 从容停止服务

    ps -ef | grep nginx 查询nginx 主进程

    killall nginx 杀死全部进程

    netstat -tlnp 查看端口号占用情况

    nginx 一些基本配置

    #运行用户，默认即是nginx，可以不进行设置
user  nginx;
#Nginx进程，一般设置为和CPU核数一样
worker_processes  1;   
#错误日志存放目录
error_log  /var/log/nginx/error.log warn;
#进程pid存放位置
pid        /var/run/nginx.pid;


events {
    worker_connections  1024; # 单个后台进程的最大并发数
}


http {
    include       /etc/nginx/mime.types;   #文件扩展名与类型映射表
    default_type  application/octet-stream;  #默认文件类型
    #设置日志模式
    log_format  main  '$remote_addr - $remote_user [$time_local] "$request" '
                      '$status $body_bytes_sent "$http_referer" '
                      '"$http_user_agent" "$http_x_forwarded_for"';

    access_log  /var/log/nginx/access.log  main;   #nginx访问日志存放位置

    sendfile        on;   #开启高效传输模式
    #tcp_nopush     on;    #减少网络报文段的数量

    keepalive_timeout  65;  #保持连接的时间，也叫超时时间

    #gzip  on;  #开启gzip压缩

    include /etc/nginx/conf.d/*.conf; #包含的子配置项位置和文件

    子文件的配置项，通过设置include子文件配置 进行端口 和文件映射路径设置
