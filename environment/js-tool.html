    # 日期格式化

    /**
 * @version 2.0
 * @description 常用时间库
 */
class DateUtils {
  localTime () {
    return this.formatTime(new Date())
  }
  // 初始化函数
  initTime (time) {
    // 非毫秒单位，且排除2020-10-2这种横杠 或者是2020/10/02 或者 2020,10,02
    const regSymbol = /-|\/|,+/g
    if (time.toString().length === 10 && !regSymbol.test(time.toString())) {
      time = time.toString().padEnd(13, '0') * 1
      time = new Date(time)
    } else {
      if (time.toString().indexOf('-') != -1) {
        time = new Date(time.replaceAll('-', '/'))
      } else {
        time = new Date(time)
      }
    }
    return time
  }
  /**
      * 格式化时间
      *
      * @param  {time} 时间
      * @param  {cFormat} 格式默认 {y}/{m}/{d} {h}:{i}:{s}
      * @return {String} 字符串
      * @example formatTime('2018-1-29', '{y}/{m}/{d} {h}:{i}:{s}') // -> 2018/01/29 00:00:00
      * @example formatTime('2020-10-20', '{y}/{m}/{d} {h}:{i}:{s}') // -> 2018/01/29 00:00:00
     */
  formatTime (time, cFormat = '{y}/{m}/{d} {h}:{i}:{s}') {
    const date = this.initTime(time)
    const formatObj = {
      y: date.getFullYear(),
      m: date.getMonth() + 1,
      d: date.getDate(),
      h: date.getHours(),
      i: date.getMinutes(),
      s: date.getSeconds(),
      a: date.getDay()
    }
    const timeStr = cFormat.replace(/{(y|m|d|h|i|s|a)+}/g, (result, key) => {
      let value = formatObj[key]
      if (key === 'a') return ['一', '二', '三', '四', '五', '六', '日'][value - 1]
      if (result.length > 0 && value < 10) {
        value = '0' + value
      }
      return value || 0
    })
    return timeStr
  }
  /**
     *
     * @param {Number/date} time 时间戳或者一个时间对象
     * @return {Number} 天数
     * @description 返回月份的天数
     * @example  getMonthOfDay('2020-10-20') =>31 getMonthOfDay('2020-2-20') =>28
     */
  getMonthOfDay (time) {
    const date = this.initTime(time)
    const year = date.getFullYear()
    const mouth = date.getMonth() + 1
    let days

    // 当月份为二月时，根据闰年还是非闰年判断天数
    if (mouth == 2) {
      days = (year % 4 == 0 && year % 100 == 0 && year % 400 == 0) || (year % 4 == 0 && year % 100 != 0) ? 28 : 29
    } else if (mouth == 1 || mouth == 3 || mouth == 5 || mouth == 7 || mouth == 8 || mouth == 10 || mouth == 12) {
      // 月份为：1,3,5,7,8,10,12 时，为大月.则天数为31；
      days = 31
    } else {
      // 其他月份，天数为：30.
      days = 30
    }
    return days
  }

  // 获取某年的第一天
  getFirstDayOfYear (time) {
    const year = this.initTime(time).getFullYear()
    return year + '-01-01 00:00:00'
  }

  // 获取某年最后一天
  getLastDayOfYear (time) {
    const year = this.initTime(time).getFullYear()
    const dateString = year + '-12-01 00:00:00'
    const endDay = this.getMonthOfDay(dateString)
    return year + '-12-' + endDay + ' 23:59:59'
  }

  // 获取某年有多少天
  getYearOfDay (time) {
    const firstDayYear = this.getFirstDayOfYear(time)
    const lastDayYear = this.getLastDayOfYear(time)
    const numSecond = (new Date(lastDayYear).getTime() - new Date(firstDayYear).getTime()) / 1000
    return Math.ceil(numSecond / (24 * 3600))
  }
  // 获取某个日期是当年中的第几天
  getDayOfYear (time) {
    const firstDayYear = this.getFirstDayOfYear(time)
    const numSecond = (new Date(time).getTime() - new Date(firstDayYear).getTime()) / 1000
    return Math.ceil(numSecond / (24 * 3600))
  }

  // 获取某个日期在这一年的第几周
  getDayOfYearWeek (time) {
    const numdays = this.getDayOfYear(time)
    return Math.ceil(numdays / 7)
  }
  // xx前 比如1分钟前 1小时前
  getTimeDiff (time) {
    const minute = 60 * 1000
    const hour = 60 * minute
    const day = 24 * hour
    const month = 31 * day
    const year = 12 * month
    time = this.initTime(time)
    const difftime = new Date().getTime() - time.getTime()
    if (difftime > year) {
      return parseInt(difftime / year) + '年前';
    }
    if (difftime > month) {
      return parseInt(difftime / month) + '个月前';
    }
    if (difftime > day) {
      return parseInt(difftime / day) + '天前';
    }
    if (difftime > hour) {
      return parseInt(difftime / hour) + '个小时前';
    }
    if (difftime > minute) {
      return parseInt(difftime / minute) + '分钟前';
    }
    return '刚刚';
  }
}

export default new DateUtils()
